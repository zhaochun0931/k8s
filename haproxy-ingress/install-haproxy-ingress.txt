helm repo add haproxytech https://haproxytech.github.io/helm-charts
helm repo update

helm install haproxy-kubernetes-ingress haproxytech/kubernetes-ingress \
  --create-namespace \
  --namespace haproxy-controller






# uninstall

helm uninstall -n haproxy-controller haproxy-kubernetes-ingress





kubectl delete deployment.apps/haproxy-kubernetes-ingress -n haproxy-controller
kubectl delete service/haproxy-kubernetes-ingress -n haproxy-controller
