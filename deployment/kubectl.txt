kubectl create deployment web-server --image=luksa/kubia:latest --replicas=3




kubectl expose deployment web-server --port=80 --target-port=8080 --name=web-server-service --type=ClusterIP

--port: The port the service will listen on (how other pods talk to it).

--target-port: The port your application is actually running on inside the container.

--type: Defines how the service is accessible (see the table below).
